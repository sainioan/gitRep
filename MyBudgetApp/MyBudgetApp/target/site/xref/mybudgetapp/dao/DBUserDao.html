8   import java.sql.Connection;
9   import java.sql.PreparedStatement;
10  import java.sql.ResultSet;
11  import java.sql.SQLException;

14  import java.util.List;
15  import mybudgetapp.domain.User;

21  public class DBUserDao implements UserDao {
22  
23      private List<User> users;
24      private String database;
25      private MyBudgetDatabase db;
26      private String password;
27  
28      String selectStmt = "SELECT * FROM user";
29  
30      public DBUserDao(MyBudgetDatabase db) {
31          this.db = db;
32  
33          users = new ArrayList<>();
34  
35          db.initializeDatabase();
36          Statement stmt = null;
37          try {
38              Connection conn = db.connect();
39              stmt = conn.createStatement();
40              ResultSet rs = stmt.executeQuery(selectStmt);
41              while (rs.next()) {
42                  User user = new User();
43                  user.setUsername(rs.getString("username"));
44                  user.setPassword(rs.getString("password"));
45                  users.add(user);
46              }
47          } catch (Exception e) {
48              System.out.println(e.getMessage());
49          }
50  
51      }
52  
53      /**
54       * The medhod saves a new user given into the database
55       *
56       * @param user new user object given as a parameter
57       * @return returns true if the saving of the user into the database succeeds
58       * @throws SQLException if saveUser fails
59       */
60      public boolean saveUser(User user) throws SQLException, Exception {
61          Connection connection = db.connect();
62          try {
63  
64              PreparedStatement saveDetailsStatement = connection.prepareStatement(
65                      "INSERT OR REPLACE INTO user (username, password) VALUES (?, ?);"
66              );
67  
68              saveDetailsStatement.setString(1, user.getUsername());
69              saveDetailsStatement.setString(2, user.getPassword());
70              saveDetailsStatement.executeUpdate();
71              saveDetailsStatement.close();
72  
73          } catch (SQLException ex) {
74              System.out.println(ex.getMessage());
75              return false;
76          }
77          return true;
78      }
79  
80      public boolean delete(String username) throws SQLException {
81          try {
82              Connection conn = db.connect();
83              PreparedStatement stmt = conn.prepareStatement("DELETE FROM user WHERE username = (?)");
84  
85              stmt.setString(1, username);
86              stmt.executeUpdate();
87              stmt.close();
88              conn.close();
89          } catch (Throwable t) {
90              System.out.println("delete user error..." + t.getMessage());
91              return false;
92          }
93          return true;
94      }

96      /**
97       * The medhod creates a new user given as a parameter by calling the
98       * saveuser method
99       *
100      * @param user user object
101      * @return returns the user object
102      * @throws SQLException if saveUser fails
103      */
104     @Override
105     public User create(User user) throws SQLException {
106 
107         users.add(user);
108 
109         try {
110             saveUser(user);
111         } catch (Exception e) {
112             System.out.println(e.getMessage());
113         }
114         return user;
115     }

117     @Override
118     public User findByUsername(String username) {
119         try {
120             for (User u : this.users) {
121                 if (u.getUsername().equals(username)) {
122                     return u;
123                 }
124             }
125         } catch (Exception e) {
126             System.out.println(e.getMessage());
127         }
128         return null;
129 
130     }
131 
132     @Override
133     public List<User> getAll() throws SQLException {
134         Connection con = db.connect();
135         PreparedStatement stmt = con.prepareStatement(selectStmt);
136         users = new ArrayList<>();
137 
138         ResultSet rs = stmt.executeQuery();
139         while (rs.next()) {
140             User user = new User(rs.getString("username"), rs.getString("password"));
141             user.setUsername(rs.getString("username"));
142             user.setPassword(rs.getString("password"));
143             users.add(user);
144         }
145         stmt.close();
146         rs.close();
147         con.close();
148         return users;
149     }

151     /**
152      * Finds a user with the given username from the database
153      *
154      * @param username name of the logged in user
155      * @return returns the user object with the username given as a parameter
156      * @throws SQLException if unable to retrieve the username from the database
157      */
158     @Override
159     public User findOne(String username) throws SQLException {
160         Connection conn = db.connect();
161         PreparedStatement stmt = conn.prepareStatement("SELECT * FROM user WHERE username = ?");
162         stmt.setString(1, username);
163 
164         ResultSet rs = stmt.executeQuery();
165         boolean findOne = rs.next();
166 
167         if (!findOne) {
168             return null;
169         }
170 
171         User user = new User(rs.getString("username"), rs.getString("password"));
172 
173         stmt.close();
174         rs.close();
175         conn.close();
176 
177         return user;
178     }
