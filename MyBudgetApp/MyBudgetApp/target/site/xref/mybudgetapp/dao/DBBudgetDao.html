100         Date sqlDate = Date.valueOf(expense.getDate());
101         try {
102             PreparedStatement statement = connection.prepareStatement(
103                     "INSERT OR REPLACE INTO expense (user_username, category_name, amount, time) VALUES (?,?,?,?);"
104             );
105             statement.setString(1, expense.getUserName());
106             statement.setString(2, expense.getCategoryName());
107             statement.setFloat(3, amountF);
108             statement.setDate(4, sqlDate);
109             statement.executeUpdate();
110             statement.close();
111         } catch (SQLException e) {

159     public Balance findOne(String username) throws SQLException {
160         Connection conn = db.connect();
161         PreparedStatement stmt = conn.prepareStatement("SELECT * FROM balance WHERE user_username = ?");
162         stmt.setString(1, username);
163         ResultSet rs = stmt.executeQuery();
164         boolean findOne = rs.next();
165         if (!findOne) {
166             return null;
167         }
168         double balanceAmount = Double.parseDouble(rs.getString("amount"));
169         Balance b = new Balance(rs.getString("user_username"), balanceAmount, LocalDate.now());
170         stmt.close();
171         rs.close();
172         conn.close();
173 
174         return b;
175     }
